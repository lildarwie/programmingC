#include <stdio.h>
#include <string.h>



int main()
{
    char str[80];
    int i;

    printf("Введи строку: ");
    fgets(str, sizeof(str), stdin);

    for (i = 0; str[i] != '\0'; i++)
    {
        if (str[i] == 'a')
            str[i] = 'A';
        else if (str[i] == 'b')
            str[i] = 'B';
    }

    printf("Изменённая строка: %s\n", str);

    return 0;
}
