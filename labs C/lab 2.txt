#include <stdio.h>
#include <math.h>

int main()
{
    int i, j, N=3;
    double summGD=0, summPD=0;
    printf("Vvedite razmer matrici: ");
    scanf("%d", &N);
    double m[N][N];
    for (i=0;i<N; i++){
        for (j=0; j<N; j++){
            printf("m[%d][%d]", i+1, j+1);
            scanf("%lf", &m[i][j]);
        }
    }
    for (i=0;i<N; i++){
        for (j=0; j<N; j++){
            printf("%lf ", m[i][j]);
        }
        printf("\n");
    }
    for (i=0;i<N; i++){
        summGD=summGD+ m[i][i];
        summPD=summPD+ m[i][N-i-1];
    }
    printf("summGD=%lf\n", summGD);
    printf("summPD=%lf\n", summPD);
    double square_m[N][N];
    for(i = 0; i < N; i++) {
        for(j = 0; j < N; j++) {
            square_m[i][j] = 0;
            for(int k = 0; k < N; k++) {
                square_m[i][j] += m[i][k] * m[k][j];
            }
        }
    }
    printf("Kvadrat matrici: \n");
    for(i = 0; i < N; i++) {
        for(j = 0; j < N; j++) {
            printf("%lf ", square_m[i][j]);
        }
        printf("\n");
    }
    return 0;
}


    getchar();
    return 0;
}
