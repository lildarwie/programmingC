#include <stdio.h>
#include <stdlib.h>
#include <math.h>

double P(double a, double b, double c)
{
    double sum = a + b + c;
    return sum;
}


double S(double a, double b, double c)
{
    double s, p = (P(a, b, c) / 2);
    s = sqrt(p * (p - a) * (p - b) * (p - c));
    return s;
}

int yes_or_no (double a, double b, double c)
{
    if (a <= 0 || b <= 0 || c <= 0) {
        return 0;
    }
    if (a + b > c && a + c > b && b + c > a) {
        return 1;
    }
    return 0;
}
